\subsubsubsection{Truy cập hệ thống}
\setlist[enumerate]{leftmargin=*,topsep=0pt,itemsep=2pt,parsep=0pt}
\setlist[itemize]{leftmargin=*,topsep=0pt,itemsep=2pt,parsep=0pt}
\begin{longtable}{|P{4.8cm}|m{10cm}|}\hline
	\texttt{Use-case ID} & UC--01\\
	\hline
	%%%%%%%%%%%%%%%
	\texttt{Use-case Name} & Truy cập hệ thống\\
	\hline
	%%%%%%%%%%%%%%%
	\texttt{Use-case Overview} & Khách truy cập hệ thống bằng tài khoản nội bộ hoặc Google OAuth; sau khi xác thực thành công, hệ thống cấp quyền theo vai trò hiện có.
	\\
	\hline
	\texttt{Actors} & Guest (Khách)
	\\
	\hline
	\texttt{Preconditions} & Khách đã mở trang đăng nhập và dịch vụ xác thực hoạt động bình thường.
	\\
	\hline
	\texttt{Trigger} & Khách chọn đăng nhập bằng tài khoản nội bộ hoặc Google.\\
	\hline
	\texttt{Basic flow} &
	\begin{enumerate}
		\item Khách truy cập trang đăng nhập.
		\item Khách chọn một trong hai cách đăng nhập: nhập tên đăng nhập/mật khẩu hoặc đăng nhập bằng Google.
		\item Nếu đăng nhập nội bộ, khách nhập thông tin tài khoản và gửi biểu mẫu.
		\item Nếu đăng nhập Google, hệ thống nhận thông tin xác thực từ Google và gửi về server để kiểm tra.
		\item Hệ thống xác thực tài khoản, kiểm tra trạng thái người dùng và lấy vai trò hiện tại.
		\item Nếu hợp lệ, hệ thống tạo phiên đăng nhập và nạp giao diện phù hợp với vai trò user hoặc admin.
	\end{enumerate}
	\\
	\hline
	\texttt{Post conditions} &
	Người dùng truy cập được hệ thống với quyền phù hợp với vai trò hiện tại.
	\\
	\hline
	\texttt{Alternative flow} &
	\begin{enumerate}[label = A\arabic*.]
		\item Ở bước 2, nếu khách chưa có tài khoản nội bộ thì chuyển sang biểu mẫu đăng ký để tạo tài khoản mới.
		\item Ở bước 5, nếu tài khoản đã liên kết Google thì hệ thống đăng nhập bằng cùng hồ sơ người dùng hiện có.
	\end{enumerate}
	\\
	\hline
	\texttt{Exception flow} &
	\begin{enumerate}[label = E\arabic*:]
		\item Ở bước 5, nếu tên đăng nhập/mật khẩu không đúng hoặc xác thực Google thất bại thì hệ thống hiển thị lỗi và giữ nguyên màn hình đăng nhập.
		\item Ở bước 5, nếu tài khoản bị chặn thì hệ thống từ chối đăng nhập và thông báo trạng thái tài khoản.
	\end{enumerate}
	\\
	\hline
	\texttt{Special requirements} & Không có.\\

	\texttt{Extension points} & Không có.\\
